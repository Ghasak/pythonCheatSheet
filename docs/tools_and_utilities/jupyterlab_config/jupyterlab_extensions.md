```sh
pipenv run jupyter lab --no-browser --allow-root --port=9999 --autoreload --notebook-dir="$(pwd)" -y
jupyter server list
jupyer lab --version
jupyter lab path
jupyter lab build
pip install --upgrade <library_name>
pip search <library_name>
```
